Code gs for App Script | Google Sheet

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  
  var row = [];
  for (var i = 0; i < headers.length; i++) {
    var field = headers[i];
    row.push(e.parameter[field] || ""); // agar field nahi bheja to blank
  }
  
  sheet.appendRow(row);
  return ContentService.createTextOutput("Success");
}

  <style>
    body {
      font-family: Arial;

    }

  </style>

<body>
  <h2>Student Form</h2>
  <form id="myForm">
    <label>Name:<br><input type="text" id="Name" name="Name" required></label><br>
    <label>Father:<br><input type="text" id="Father" name="Father" required></label><br>
    <label>Phone:<br><input type="text" id="Phone" name="Phone" required></label><br>
    <label>Email:<br><input type="email" id="Email" name="Email" required></label><br>
    <label>Address:<br><input type="text" id="Address" name="Address" required></label><br><br>
    <button type="submit">Submit</button>
  </form>

  <script>
    const scriptURL = 'https://script.google.com/macros/s/AKfycbxFnyTIPoQVVQlmnruhBrZw-tbQT5P_sBBPY7DD-OZPPOjk6byebmO78kMydjxzs_CYuw/exec'; // <-- yahan apna URL daalo
    const form = document.getElementById('myForm');

    form.addEventListener('submit', function (e) {
      e.preventDefault();

      Swal.fire({
        title: 'Are you sure?',
        text: 'Do you want to submit this form?',
        icon: 'question',
        showCancelButton: true,
        confirmButtonText: 'Yes, submit it!',
        cancelButtonText: 'Cancel'
      }).then((result) => {
        if (result.isConfirmed) {
          Swal.fire({
            title: 'Submitting...',
            html: 'Please wait',
            allowOutsideClick: false,
            didOpen: () => {
              Swal.showLoading();
            }
          });

          const formData = new FormData(form);
          fetch(scriptURL, { method: 'POST', body: formData })
            .then(response => {
              Swal.fire({
                title: 'Success!',
                text: 'Your form has been submitted.',
                icon: 'success',
              });
              form.reset();
            })
            .catch(error => {
              Swal.fire('Error!', error.message, 'error');
            });
        }
      });
    });
  </script>
</body>
